from sqlalchemy.sql import table, column, select

t = table('t', column('x'))

s = select([t]).where(t.c.x == 5)

# compile_kwargs에 해당 parameter를 넘겨주지 않을 경우, 일부 변수에 대해 치환되지 않은 Query를 Return 한다
print(s.compile(compile_kwargs={"literal_binds": True}))
